# NEW function, added to graphify/serve.py at module level (alongside _load_dedup_report etc.):
def _load_enrichment_overlay(G, out_dir: Path) -> None:
    """Merge enrichment.json overlay onto G in-place. ENRICH-08 + D-06 augmentation.

    Reads `out_dir/enrichment.json`; for each present pass, sets enrichment-specific
    attributes on nodes WITHOUT overwriting existing ones:

      - passes.description → G.nodes[canonical_nid]["enriched_description"] = sanitized_text
      - passes.community   → G.nodes[canonical_nid]["community_summary"] = sanitized_text  (for each node in community)
      - passes.staleness   → G.nodes[canonical_nid]["staleness_override"] = "FRESH"|"STALE"|"GHOST"
      - passes.patterns    → G.graph["patterns"] = list[...]  (graph-level, not per-node)

    D-16 ENRICH-12: each `<node_id>` key from enrichment.json is routed through
    `_load_dedup_report(out_dir)`'s alias_map before `G.nodes[...]` lookup.

    Idempotent: repeated calls with the same enrichment.json produce identical G state.
    Never raises: missing file → no-op; malformed → logged to stderr + no-op
    (uses enrich._validate_enrichment_envelope to guard).
    """

# Closure state added to serve():
# _enrichment_mtime: float = 0.0  # alongside existing _graph_mtime

# Extended _reload_if_stale inside serve():
def _reload_if_stale() -> None:
    """Reload G and re-apply enrichment overlay when either sidecar mtime changes.
    graph.json change → reload G + re-apply overlay.
    enrichment.json change → just re-apply overlay (G unchanged).
    """
    nonlocal G, communities, _graph_mtime, _branching_factor, _enrichment_mtime
    # ---- existing graph.json path (unchanged) ----
    try:
        mtime = os.stat(graph_path).st_mtime
    except OSError:
        return
    if mtime != _graph_mtime:
        G = _load_graph(graph_path)
        communities = _communities_from_graph(G)
        _branching_factor = _compute_branching_factor(G)
        _graph_mtime = mtime
        # After reloading graph, always re-apply enrichment overlay
        _load_enrichment_overlay(G, _out_dir)
        # Refresh enrichment mtime tracker too
        try:
            _enrichment_mtime = os.stat(_out_dir / "enrichment.json").st_mtime
        except OSError:
            _enrichment_mtime = 0.0
    else:
        # ---- NEW enrichment.json path (ENRICH-09) ----
        try:
            emtime = os.stat(_out_dir / "enrichment.json").st_mtime
        except OSError:
            return
        if emtime != _enrichment_mtime:
            _load_enrichment_overlay(G, _out_dir)
            _enrichment_mtime = emtime
```

    def _load_enrichment_overlay(G, out_dir: Path) -> None:
        """Merge enrichment.json overlay onto G in-place. ENRICH-08 + D-06 augmentation.

        Reads <out_dir>/enrichment.json; for each present pass, sets NEW node attributes
        (never overwriting base fields). Silent on missing or malformed file.
        D-16 ENRICH-12: node_ids routed through dedup alias_map on read.
        """
        import json as _json
        from pathlib import Path as _Path
        from graphify.enrich import _validate_enrichment_envelope

        p = _Path(out_dir) / "enrichment.json"
        try:
            if not p.exists():
                return
            data = _json.loads(p.read_text(encoding="utf-8"))
        except (OSError, _json.JSONDecodeError) as exc:
            print(f"[graphify] serve: enrichment.json unreadable: {exc}", file=sys.stderr)
            return
        if not _validate_enrichment_envelope(data):
            # _validate_enrichment_envelope already logged the reason
            return

        alias_map = _load_dedup_report(_Path(out_dir))  # {eliminated_id: canonical_id}
        passes = data.get("passes", {})

        # passes.description → G.nodes[canonical]["enriched_description"]
        for nid, text in passes.get("description", {}).items():
            canonical = alias_map.get(nid, nid)
            if canonical in G.nodes:
                G.nodes[canonical]["enriched_description"] = text

        # passes.community → per-community summary applied to every node in that community
        communities_map = passes.get("community", {})
        if communities_map:
            # Find which community each node belongs to via G.nodes[nid].get("community")
            for nid, nattrs in G.nodes(data=True):
                cid = nattrs.get("community")
                if cid is None:
                    continue
                summary = communities_map.get(str(cid))
                if summary:
                    nattrs["community_summary"] = summary

        # passes.staleness → G.nodes[canonical]["staleness_override"]
        for nid, label in passes.get("staleness", {}).items():
            canonical = alias_map.get(nid, nid)
            if canonical in G.nodes:
                G.nodes[canonical]["staleness_override"] = label

        # passes.patterns → G.graph["patterns"]  (graph-level attribute)
        patterns = passes.get("patterns", [])
        if patterns:
            # Thread each pattern's nodes list through alias_map for consistency
            resolved = []
            for p_entry in patterns:
                if not isinstance(p_entry, dict):
                    continue
                nodes_list = p_entry.get("nodes", []) or []
                resolved_nodes = [alias_map.get(n, n) for n in nodes_list]
                resolved.append({**p_entry, "nodes": resolved_nodes})
            G.graph["patterns"] = resolved
    ```
