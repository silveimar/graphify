    """Grep-CI whitelist guard: only build.py + __main__.py + watch.py may write graph.json.

    SC-5 structural enforcement of v1.1 D-invariant (Pitfall 3 mitigation).
    """
    from pathlib import Path
    import re

    _GRAPHIFY_DIR = Path(__file__).resolve().parent.parent / "graphify"
    _ALLOWED = {"build.py", "__main__.py", "watch.py"}

    def test_to_json_caller_whitelist():
        """Only allowed files may contain the literal token `to_json(` as a call-site.
        Imports (`from graphify.export import to_json`) are fine; CALLS are the invariant."""
        offenders: list[str] = []
        pattern = re.compile(r"\\bto_json\\s*\\(")
        for py_file in _GRAPHIFY_DIR.glob("*.py"):
            if py_file.name in _ALLOWED:
                continue
            src = py_file.read_text(encoding="utf-8")
            # Strip imports; look only at call sites (heuristic: lines not starting with `from` or `import`)
            call_lines = [ln for ln in src.splitlines()
                          if pattern.search(ln) and not ln.strip().startswith(("from ", "import "))]
            if call_lines:
                offenders.append(f"{py_file.name}: {call_lines[:3]}")
        assert not offenders, (
            "SC-5 violation — only build.py, __main__.py, watch.py may CALL to_json(). "
            f"Offenders: {offenders}. If a new call-site is needed, update the whitelist "
            "AND get approval from the Phase 15 invariant-owner."
        )

    def test_enrich_py_never_imports_to_json():
        """enrich.py is structurally forbidden from importing to_json."""
        src = (_GRAPHIFY_DIR / "enrich.py").read_text(encoding="utf-8")
        assert "to_json" not in src, (
            "enrich.py must NEVER reference to_json — graph.json is read-only to Phase 15. "
            "Overlay data goes to enrichment.json only."
        )

    def test_enrich_py_never_writes_graph_json():
        """enrich.py may not contain the literal string 'graph.json' on the write side.
        (It may READ graph.json — in _load_enrichment_overlay's consumer flow — but no writes.)"""
        src = (_GRAPHIFY_DIR / "enrich.py").read_text(encoding="utf-8")
        assert "_write_graph_json" not in src
        # Write-pattern sentinel: no `os.replace(... , "graph.json")` or similar
        write_patterns = [r'\\.write_text\\s*\\([^)]*graph\\.json', r'os\\.replace\\s*\\([^,]*,[^)]*graph\\.json']
        for pat in write_patterns:
            assert not re.search(pat, src), (
                f"enrich.py contains a graph.json write pattern (regex {pat!r}). "
                "Violates Phase 15 v1.1 D-invariant."
            )
