    ```python
    """Doc-fence-as-fixture tests for docs/TEMPLATES.md.

    Each section locked in CONTEXT.md D-55.11 has at least one annotated
    ```template <!-- test:<label> --> fence in docs/TEMPLATES.md. This module
    extracts those fences and runs them through _expand_blocks to ensure the
    doc cannot rot against the engine (D-55.12).
    """
    from __future__ import annotations

    import re
    from pathlib import Path

    import networkx as nx
    import pytest

    from graphify.templates import (
        BlockContext,
        _build_edge_records,
        _expand_blocks,
    )

    TEMPLATES_MD = Path(__file__).resolve().parent.parent / "docs" / "TEMPLATES.md"


    def _extract_fence(label: str) -> str:
        """Extract a ```template fence annotated with <!-- test:<label> -->."""
        if not TEMPLATES_MD.exists():
            pytest.fail(f"docs/TEMPLATES.md does not exist (Plan 55-05 must land first)")
        text = TEMPLATES_MD.read_text(encoding="utf-8")
        pattern = rf"```template <!-- test:{re.escape(label)} -->\n(.*?)```"
        m = re.search(pattern, text, re.DOTALL)
        assert m, f"No annotated fence with label {label!r} found in TEMPLATES.md"
        return m.group(1).rstrip("\n")


    def _make_thing_ctx(**attrs):
        G = nx.Graph()
        G.add_node("n", label="N", **attrs)
        return BlockContext(
            graph=G,
            node_id="n",
            edges=_build_edge_records(G, "n"),
            dataview_nonempty=False,
            note_type="thing",
        )


    # ------------------------------------------------------------------ section 1
    def test_doc_fence_conditional_basic_renders():
        text = _extract_fence("conditional-basic")
        ctx = _make_thing_ctx()
        # Each fence ships with its own assertion contract documented in TEMPLATES.md
        # under D-55.12. This test asserts the fence parses and renders without error.
        out = _expand_blocks(text, ctx)
        assert isinstance(out, str)
        assert "{{#" not in out, "Block markers must be fully expanded"


    # ------------------------------------------------------------------ section 2
    def test_doc_fence_connections_loop_renders():
        text = _extract_fence("connections-loop")
        G = nx.Graph()
        G.add_node("n", label="N")
        G.add_node("m", label="M")
        G.add_edge("n", "m", relation="references")
        ctx = BlockContext(
            graph=G, node_id="n",
            edges=_build_edge_records(G, "n"),
            dataview_nonempty=False, note_type="thing",
        )
        out = _expand_blocks(text, ctx)
        assert "{{#connections}}" not in out
        assert "{{/connections}}" not in out


    # ------------------------------------------------------------------ section 3
    def test_doc_fence_ordering_invariant_renders():
        # Ordering invariant: blocks expand BEFORE ${} substitution (D-16).
        text = _extract_fence("ordering-invariant")
        ctx = _make_thing_ctx()
        out = _expand_blocks(text, ctx)
        assert "{{#" not in out


    # ------------------------------------------------------------------ section 4
    def test_doc_fence_sanitization_renders():
        # Sanitization sinks: label/HTML escape contract (T-31-01).
        text = _extract_fence("sanitization")
        G = nx.Graph()
        G.add_node("n", label="N")
        G.add_node("evil", label="<script>alert(1)</script>")
        G.add_edge("n", "evil", relation="references")
        ctx = BlockContext(
            graph=G, node_id="n",
            edges=_build_edge_records(G, "n"),
            dataview_nonempty=False, note_type="thing",
        )
        out = _expand_blocks(text, ctx)
        assert "<script>" not in out, "Label injection must be sanitized"


    # ------------------------------------------------------------------ section 5
    def test_doc_fence_predicate_catalog_renders():
        text = _extract_fence("predicate-catalog")
        ctx = _make_thing_ctx()
        out = _expand_blocks(text, ctx)
        assert "{{#" not in out


    # ------------------------------------------------------------------ section 6
    def test_doc_fence_predicate_flags_renders():
        text = _extract_fence("predicate-flags")
        # Build a ctx with a registered flag handler matching the doc example
        ctx = _make_thing_ctx(is_published=True)
        # Doc example expected to use {{#if_flag_is_published}}; supply matching handler
        ctx_with_flags = BlockContext(
            graph=ctx.graph,
            node_id=ctx.node_id,
            edges=ctx.edges,
            dataview_nonempty=False,
            note_type="thing",
            flag_predicates={
                "is_published": lambda c: bool(c.graph.nodes[c.node_id].get("is_published")),
            },
        )
        out = _expand_blocks(text, ctx_with_flags)
        assert "{{#" not in out


    # ------------------------------------------------------------------ section 7
    def test_doc_fence_validation_renders():
        text = _extract_fence("validation")
        ctx = _make_thing_ctx()
        out = _expand_blocks(text, ctx)
        assert "{{#" not in out


    # ------------------------------------------------------------------ section 8
    def test_doc_fence_backward_compat_renders():
        # Block-free templates remain byte-identical (ROADMAP-31 criterion 4).
        text = _extract_fence("backward-compat")
        ctx = _make_thing_ctx()
        out = _expand_blocks(text, ctx)
        assert "{{#" not in out
    ```

    Notes:
    - Each test is intentionally permissive on the exact rendered output (asserts only that block markers are gone + sanitization holds) so the docs author has freedom to choose example wording. The doc is the source of truth; the test just gates that the example *runs*.
    - The `_extract_fence` helper FAILs (not errors) when TEMPLATES.md is absent, with a message pointing at Plan 55-05 — clear signal during Wave 2 RED state.
    - No new dependencies; uses stdlib re/pathlib + existing networkx.
