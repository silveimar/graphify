       ```python
       def _compile_flag_predicates(profile: dict | None) -> dict:
           """Compile profile['predicate_flags'] into a {name: callable(ctx) -> bool} dict.

           Two-shape schema (D-55.07):
             {attr: <name>}              → bool(node.get(<name>))
             {attr: <name>, equals: <v>} → node.get(<name>) == v
           Validation lives in profile._validate_predicate_flags; this helper trusts preflight.
           """
           if not profile:
               return {}
           flags = profile.get("predicate_flags") or {}
           if not isinstance(flags, dict):
               return {}
           compiled: dict = {}
           for name, rule in flags.items():
               if not isinstance(rule, dict):
                   continue
               attr = rule.get("attr")
               if not isinstance(attr, str) or not attr:
                   continue
               if "equals" in rule:
                   expected = rule["equals"]
                   def _eq_handler(ctx, _attr=attr, _expected=expected):
                       node = ctx.graph.nodes.get(ctx.node_id, {})
                       return node.get(_attr) == _expected
                   compiled[name] = _eq_handler
               else:
                   def _truthy_handler(ctx, _attr=attr):
                       node = ctx.graph.nodes.get(ctx.node_id, {})
                       return bool(node.get(_attr))
                   compiled[name] = _truthy_handler
           return compiled
       ```
       Default-arg trick (`_attr=attr`) freezes the closure variable — required because Python late-binds in for loops.

       ```python
       def _validate_predicate_flags(flags, prefix: str = "predicate_flags") -> list[str]:
           """Validate profile['predicate_flags']. Returns list of error strings — empty = valid."""
           # Local import to avoid templates.py ↔ profile.py cycle (mirrors L1447 precedent)
           from graphify.templates import _PREDICATE_CATALOG

           errors: list[str] = []
           if flags is None:
               return errors
           if not isinstance(flags, dict):
               errors.append(
                   f"{prefix}: must be a dict, got {type(flags).__name__}"
               )
               return errors
           if len(flags) > _PREDICATE_FLAGS_MAX_ENTRIES:
               errors.append(
                   f"{prefix}: too many entries ({len(flags)}); cap is "
                   f"{_PREDICATE_FLAGS_MAX_ENTRIES} (T-55-FLAG-VAL)"
               )
               return errors
           for name, rule in flags.items():
               if not isinstance(name, str) or not _IF_FLAG_NAME_RE.match(name):
                   errors.append(
                       f"{prefix}: flag name {name!r} must match /^[a-z][a-z0-9_]*$/"
                   )
                   continue
               # Catalog collision (D-55.08): user names "god_node" → "if_god_node" already in catalog
               if f"if_{name}" in _PREDICATE_CATALOG:
                   errors.append(
                       f"{prefix}.{name}: name collides with built-in predicate "
                       f"'if_{name}' — choose a different name"
                   )
                   continue
               # if_attr_ prefix collision (D-55.08)
               if name.startswith("attr_"):
                   errors.append(
                       f"{prefix}.{name}: name starting with 'attr_' is reserved for "
                       "the {{#if_attr_<name>}} escape hatch — choose a different name"
                   )
                   continue
               # Rule shape (mirrors mapping.py:129-135)
               if not isinstance(rule, dict):
                   errors.append(f"{prefix}.{name}: rule must be a dict, got {type(rule).__name__}")
                   continue
               attr = rule.get("attr")
               if not isinstance(attr, str) or not attr:
                   errors.append(f"{prefix}.{name}.attr: required, must be a non-empty string")
               extra = set(rule) - {"attr", "equals"}
               if extra:
                   errors.append(
                       f"{prefix}.{name}: unknown keys {sorted(extra)} — "
                       "only 'attr' and 'equals' are supported"
                   )
           return errors
       ```
