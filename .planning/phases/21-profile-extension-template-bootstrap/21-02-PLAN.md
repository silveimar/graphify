    ```python
    """Excalidraw template stub renderer and writer (Phase 21, TMPL-01..05)."""
    from __future__ import annotations
    import json
    from pathlib import Path
    from graphify.profile import validate_vault_path, safe_frontmatter_value

    SCENE_JSON_SKELETON = {
        "type": "excalidraw",
        "version": 2,
        "source": "graphify",
        "elements": [],
        "appState": {
            "viewBackgroundColor": "#ffffff",
            "gridSize": None,
            "currentItemFontFamily": 5,
        },
        "files": {},
    }

    def render_stub(diagram_type: dict) -> str:
        """Render a single .excalidraw.md stub string."""
        name = safe_frontmatter_value(str(diagram_type.get("name", "diagram")))
        scene_json = json.dumps(SCENE_JSON_SKELETON, indent=2)
        return (
            "---\n"
            "excalidraw-plugin: parsed\n"
            "compress: false\n"
            f"tags: [excalidraw, graphify, {name}]\n"
            "---\n\n"
            "## Text Elements\n\n"
            "## Drawing\n"
            "```json\n"
            f"{scene_json}\n"
            "```\n"
        )

    def write_stubs(vault_dir, diagram_types, force: bool = False) -> list[Path]:
        """Write one .excalidraw.md stub per diagram_type. Idempotent unless force=True."""
        vault_dir = Path(vault_dir)
        written: list[Path] = []
        for dt in diagram_types:
            rel = dt.get("template_path") or f"Excalidraw/Templates/{dt['name']}.excalidraw.md"
            # Path confinement (T-21-01, T-21-02 mitigation)
            target = validate_vault_path(rel, vault_dir)
            target.parent.mkdir(parents=True, exist_ok=True)
            if target.exists() and not force:
                continue
            target.write_text(render_stub(dt), encoding="utf-8")
            written.append(target)
        return written
    ```

    if cmd == "--init-diagram-templates":
        from graphify.excalidraw import write_stubs
        from graphify.profile import load_profile
        vault = None
        force = False
        i = 0
        while i < len(args):
            if args[i] == "--vault":
                vault = args[i + 1]; i += 2; continue
            if args[i] == "--force":
                force = True; i += 1; continue
            print(f"error: unknown --init-diagram-templates option: {args[i]}", file=sys.stderr)
            sys.exit(2)
        if not vault:
            print("error: --vault PATH required", file=sys.stderr); sys.exit(2)
        profile = load_profile(vault)
        diagram_types = profile.get("diagram_types") or []
        written = write_stubs(vault, diagram_types, force=force)
        print(f"[graphify] init-diagram-templates complete: wrote {len(written)} stub(s) (force={force})")
        return

    ```python
    """Phase 21 TMPL-06: grep-scope denylist for direct vault .md writes."""
    import re
    from pathlib import Path

    FORBIDDEN_PATTERNS = [
        r"\.write_text\(",       # Path.write_text on .md
        r"write_note_directly",
        r"open\(['\"][^'\"]*\.md['\"],\s*['\"]w",  # open('xxx.md', 'w')
    ]

    SCANNED = ["graphify/seed.py", "graphify/export.py", "graphify/__main__.py"]

    def test_no_direct_md_writes_in_seed_export_main():
        repo = Path(__file__).resolve().parent.parent
        violations = []
        for rel in SCANNED:
            src = (repo / rel).read_text(encoding="utf-8")
            for pat in FORBIDDEN_PATTERNS:
                for m in re.finditer(pat, src):
                    line_no = src[:m.start()].count("\n") + 1
                    line = src.splitlines()[line_no - 1]
                    # Allow writes targeting Excalidraw/Templates/*.excalidraw.md context
                    if "Excalidraw/Templates" in line or "excalidraw.md" in line:
                        continue
                    # Allow non-.md write_text (e.g., .json, .yaml, .html)
                    if pat == r"\.write_text\(" and not re.search(r"\.md['\"]|\.md\s*\)", line):
                        continue
                    violations.append(f"{rel}:{line_no}: {line.strip()}")
        assert not violations, (
            "Direct vault .md write detected. Route through graphify.merge.compute_merge_plan "
            "or graphify.excalidraw.write_stubs (Excalidraw/Templates/ only). Violations:\n"
            + "\n".join(violations)
        )

    def test_no_lzstring_import_anywhere():
        repo = Path(__file__).resolve().parent.parent
        for py in (repo / "graphify").rglob("*.py"):
            src = py.read_text(encoding="utf-8")
            assert "lzstring" not in src.lower() or "# noqa" in src, (
                f"{py}: lzstring import forbidden — compress: false one-way door (D-{py.name})"
            )

    def test_only_merge_compute_merge_plan_writes_vault_md():
        """Sanity: graphify.merge exposes compute_merge_plan as the sanctioned writer."""
        from graphify.merge import compute_merge_plan
        assert callable(compute_merge_plan)
    ```
