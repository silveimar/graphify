    ```python
    def _run_list_diagram_seeds_core(
        G: "nx.Graph",
        project_root: "Path",
        arguments: dict,
        alias_map: dict[str, str] | None = None,
    ) -> str:
        """SEED-09. Never raises. Always returns D-02 envelope."""
        budget = int(arguments.get("budget", 500))
        seeds_dir = project_root / "graphify-out" / "seeds"
        # Prefer reading manifest for summary; fall back to globbing *.json if manifest missing.
        manifest_path = seeds_dir / "seeds-manifest.json"

        def _no_seeds() -> str:
            meta_nc = {"status": "no_seeds", "seed_count": 0, "budget_used": 0}
            return "" + QUERY_GRAPH_META_SENTINEL + json.dumps(meta_nc, ensure_ascii=False)

        if not seeds_dir.exists():
            return _no_seeds()

        # Alias threading (D-16)
        _resolved_aliases: dict[str, list[str]] = {}
        _effective_alias_map: dict[str, str] = alias_map or {}
        def _resolve_alias(node_id: str) -> str:
            canonical = _effective_alias_map.get(node_id)
            if canonical and canonical != node_id:
                aliases = _resolved_aliases.setdefault(canonical, [])
                if node_id not in aliases:
                    aliases.append(node_id)
                return canonical
            return node_id

        # Read manifest (tolerate corruption per SP-8)
        manifest_entries: list[dict] = []
        if manifest_path.exists():
            try:
                manifest_entries = json.loads(manifest_path.read_text(encoding="utf-8")) or []
            except (json.JSONDecodeError, OSError):
                print("[graphify] seeds-manifest.json unreadable — using directory scan", file=sys.stderr)
                manifest_entries = []

        # Build summaries by reading each seed file (skip dropped-by-cap entries)
        lines: list[str] = []
        seed_count = 0
        for entry in manifest_entries:
            if entry.get("dropped_due_to_cap"):
                continue
            seed_file = seeds_dir / entry.get("seed_file", "")
            if not seed_file.exists():
                continue
            try:
                seed = json.loads(seed_file.read_text(encoding="utf-8"))
            except (json.JSONDecodeError, OSError):
                continue
            canonical_id = _resolve_alias(seed.get("seed_id", ""))
            line = "\t".join([
                canonical_id,
                str(seed.get("main_node_label", "")),
                str(seed.get("suggested_layout_type", "")),
                str(seed.get("trigger", "")),
                str(len(seed.get("main_nodes", [])) + len(seed.get("supporting_nodes", []))),
            ])
            lines.append(line)
            seed_count += 1
            if sum(len(l) for l in lines) > budget * 200:  # rough character cap
                break

        if seed_count == 0:
            return _no_seeds()

        meta: dict = {"status": "ok", "seed_count": seed_count, "budget_used": sum(len(l) for l in lines)}
        if _resolved_aliases:
            meta["resolved_from_alias"] = _resolved_aliases
        return "\n".join(lines) + QUERY_GRAPH_META_SENTINEL + json.dumps(meta, ensure_ascii=False)
    ```

    ```python
    def _run_get_diagram_seed_core(
        G: "nx.Graph",
        project_root: "Path",
        arguments: dict,
        alias_map: dict[str, str] | None = None,
    ) -> str:
        """SEED-10. Never raises. Always returns D-02 envelope."""
        requested_id_raw = str(arguments.get("seed_id", ""))
        budget = int(arguments.get("budget", 2000))
        seeds_dir = project_root / "graphify-out" / "seeds"

        # D-16 alias threading on seed_id arg
        _resolved_aliases: dict[str, list[str]] = {}
        _effective_alias_map: dict[str, str] = alias_map or {}
        def _resolve_alias(node_id: str) -> str:
            canonical = _effective_alias_map.get(node_id)
            if canonical and canonical != node_id:
                aliases = _resolved_aliases.setdefault(canonical, [])
                if node_id not in aliases:
                    aliases.append(node_id)
                return canonical
            return node_id

        canonical_id = _resolve_alias(requested_id_raw)

        def _not_found() -> str:
            meta_nf = {"status": "not_found", "seed_id": requested_id_raw, "budget_used": 0}
            if _resolved_aliases:
                meta_nf["resolved_from_alias"] = _resolved_aliases
            return "" + QUERY_GRAPH_META_SENTINEL + json.dumps(meta_nf, ensure_ascii=False)

        if not canonical_id or not seeds_dir.exists():
            return _not_found()

        seed_path = seeds_dir / f"{canonical_id}-seed.json"
        if not seed_path.exists():
            return _not_found()

        try:
            seed = json.loads(seed_path.read_text(encoding="utf-8"))
        except (json.JSONDecodeError, OSError):
            meta_c = {"status": "corrupt", "seed_id": requested_id_raw, "budget_used": 0}
            return "" + QUERY_GRAPH_META_SENTINEL + json.dumps(meta_c, ensure_ascii=False)

        # Also resolve node IDs inside the seed body
        for collection in ("main_nodes", "supporting_nodes"):
            for n in seed.get(collection, []):
                if isinstance(n, dict) and "id" in n:
                    n["id"] = _resolve_alias(n["id"])
        for rel in seed.get("relations", []):
            if isinstance(rel, dict):
                if "source" in rel:
                    rel["source"] = _resolve_alias(rel["source"])
                if "target" in rel:
                    rel["target"] = _resolve_alias(rel["target"])

        text_body = json.dumps(seed, indent=2, ensure_ascii=False)
        node_count = len(seed.get("main_nodes", [])) + len(seed.get("supporting_nodes", []))
        meta: dict = {
            "status": "ok",
            "seed_id": canonical_id,
            "node_count": node_count,
            "budget_used": len(text_body),
        }
        if _resolved_aliases:
            meta["resolved_from_alias"] = _resolved_aliases
        return text_body + QUERY_GRAPH_META_SENTINEL + json.dumps(meta, ensure_ascii=False)
    ```
