       ```python
       def _calibration_histogram(inf_scores: list[float]) -> list[int]:
           bins = [0] * 10

               bins[idx] += 1

       def _calibration_flags(bins: list[int], scores: list[float]) -> list[tuple[str, float, float]]:
           flags: list[tuple[str, float, float]] = []
           total = sum(bins) or 1
           # Mode collapse: any single bin > threshold of total
           max_share = max(bins) / total
           if max_share > _CALIBRATION_MODE_COLLAPSE_THRESHOLD:
               flags.append(("mode_collapse", round(max_share, 3), _CALIBRATION_MODE_COLLAPSE_THRESHOLD))
           # Refusal: > threshold of scores exactly 0.5
           refusal_share = sum(1 for s in scores if s == 0.5) / total
           if refusal_share > _CALIBRATION_REFUSAL_THRESHOLD:
               flags.append(("refusal", round(refusal_share, 3), _CALIBRATION_REFUSAL_THRESHOLD))
           # No-negatives: < floor share of scores below 0.5
           neg_share = sum(1 for s in scores if s < 0.5) / total
           if neg_share < _CALIBRATION_NEGATIVE_FLOOR:
               flags.append(("no_negatives", round(neg_share, 3), _CALIBRATION_NEGATIVE_FLOOR))
           return flags
       ```
