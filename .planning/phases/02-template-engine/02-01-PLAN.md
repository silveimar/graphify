    3. Add a new public helper function at the bottom of the file, after `safe_filename`:
       ```python
       def _dump_frontmatter(fields: dict) -> str:
           """Emit YAML frontmatter block (including --- delimiters) from a dict.

           Preserves insertion order. Skips None values. Uses type dispatch
           so dates/floats/bools/ints render as correct YAML types in
           Obsidian Properties (not string-quoted scalars).
           """
           lines: list[str] = ["---"]
           for key, value in fields.items():
               if value is None:
                   continue
               if isinstance(value, list):
                   lines.append(f"{key}:")
                   for item in value:
                       lines.append(f"  - {safe_frontmatter_value(str(item))}")
               elif isinstance(value, bool):
                   lines.append(f"{key}: {'true' if value else 'false'}")
               elif isinstance(value, int):
                   lines.append(f"{key}: {value}")
               elif isinstance(value, float):
                   # Note: bool is checked before int (critical for YAML); float vs bool
                   # order is irrelevant because isinstance(True, float) is False.
                   lines.append(f"{key}: {value:.2f}")
               elif isinstance(value, datetime.date):
                   lines.append(f"{key}: {value.isoformat()}")
               else:
                   lines.append(f"{key}: {safe_frontmatter_value(str(value))}")
           lines.append("---")
           return "\n".join(lines)
       ```

       CRITICAL order: `bool` must be checked BEFORE `int` because `bool` is a subclass of `int` in Python and `isinstance(True, int)` is `True`. Getting this wrong makes `True` render as `1`.
