    ```python
    def _redact_secrets(value: str) -> tuple[str, list[str]]:
        """Return (cleaned_value, matched_pattern_names)."""
        matches: list[str] = []
        cleaned = value
        for name, pattern in SECRET_PATTERNS:
            if pattern.search(cleaned):
                matches.append(name)
                cleaned = pattern.sub(_REDACTION_MARKER, cleaned)
        return cleaned, matches


    def scan_annotations_for_secrets(
        annotations: list[dict[str, Any]],
        *,
        mode: str = "redact",
    ) -> tuple[list[dict[str, Any]], list[dict[str, Any]]]:
        """Scan every string value in annotation fields NOT in ANNOTATION_ALLOW_LIST.

        Args:
            annotations: list of annotation dicts (as loaded by _load_sidecars).
            mode: 'redact' replaces matches with [REDACTED]; 'error' raises
                  ValueError listing offending annotation ids.

        Returns:
            (cleaned_annotations, findings) where findings is a list of
            {'id': annotation_id, 'field': key, 'patterns': [name, ...]}.
        """
        if mode not in {"redact", "error"}:
            raise ValueError(f"unknown secrets_mode: {mode!r}")

        cleaned: list[dict[str, Any]] = []
        findings: list[dict[str, Any]] = []
        for a in annotations:
            new_a: dict[str, Any] = {}
            ann_id = str(a.get("id") or "<unknown>")
            for k, v in a.items():
                if k in ANNOTATION_ALLOW_LIST or not isinstance(v, str):
                    new_a[k] = v
                    continue
                redacted, matched = _redact_secrets(v)
                new_a[k] = redacted
                if matched:
                    findings.append(
                        {"id": ann_id, "field": k, "patterns": matched}
                    )
            cleaned.append(new_a)

        if mode == "error" and findings:
            ids = sorted({f["id"] for f in findings})
            raise ValueError(
                "harness export: secret patterns detected in annotations; "
                f"offending annotation ids: {ids}. "
                "Re-run with --secrets-mode redact to continue."
            )
        return cleaned, findings
    ```

    **Add fidelity helpers:**
    ```python
    import hashlib

    def _sha256_file(path: Path) -> str:
        h = hashlib.sha256()
        h.update(path.read_bytes())
        return h.hexdigest()


    def _write_fidelity_manifest(
        harness_dir: Path,
        written: list[Path],
        *,
        prior: dict[str, Any] | None = None,
    ) -> Path:
        """HARNESS-08: record per-file SHA-256 + byte-length.

        If `prior` is provided (previous fidelity.json contents) and every file
        matches byte-for-byte, set `round_trip: 'byte-equal'`. Otherwise
        `round_trip: 'first-export'` (no prior) or `round_trip: 'drift'` (mismatch).
        """
        files: dict[str, dict[str, Any]] = {}
        for p in written:
            files[p.name] = {
                "sha256": _sha256_file(p),
                "bytes": p.stat().st_size,
            }

        round_trip = "first-export"
        if prior is not None:
            prior_files = prior.get("files") or {}
            if (
                prior_files
                and all(
                    prior_files.get(name, {}).get("sha256") == meta["sha256"]
                    and prior_files.get(name, {}).get("bytes") == meta["bytes"]
                    for name, meta in files.items()
                )
                and set(prior_files.keys()) == set(files.keys())
            ):
                round_trip = "byte-equal"
            else:
                round_trip = "drift"

        manifest = {
            "version": 1,
            "target": "claude",
            "round_trip": round_trip,
            "files": files,
        }

        # Mirror capability.py::write_manifest_atomic pattern
        out = harness_dir / "fidelity.json"
        tmp = out.with_suffix(".tmp")
        payload = json.dumps(manifest, indent=2, sort_keys=True, ensure_ascii=False) + "\n"
        tmp.write_text(payload, encoding="utf-8")
        os.replace(tmp, out)
        return out
    ```
