    ```python
    _CONFIDENCE_BAND_CUTPOINTS = {
        "high":   (0.8, 1.0 + 1e-9),  # x >= 0.8
        "medium": (0.5, 0.8),          # 0.5 <= x < 0.8
        "low":    (0.0, 0.5),          # x < 0.5
    }

    def _validate_relations_filter_arg(value):
        """CQUERY (Phase 67, D-12) relations whitelist for concept_code_hops.
        Accepts None (no filter), [] (zero-match), or list[str].
        Distinct from legacy _validate_relations_arg at line 2246, which rejects [].
        """
        if value is None:
            return None
        if not isinstance(value, list):
            raise ValueError("relations filter must be a list of strings or None")
        for item in value:
            if not isinstance(item, str):
                raise ValueError(
                    f"relations filter members must be strings, got {type(item).__name__}"
                )
        return list(value)

    def _resolve_confidence_band(band):
        """Phase 67 D-10 cutpoints: high >= 0.8, medium 0.5 <= x < 0.8, low < 0.5."""
        if band is None:
            return None
        if band not in _CONFIDENCE_BAND_CUTPOINTS:
            raise ValueError(
                f"confidence_band must be one of {sorted(_CONFIDENCE_BAND_CUTPOINTS)}, got {band!r}"
            )
        return _CONFIDENCE_BAND_CUTPOINTS[band]

    def _build_concept_hops_filter(min_confidence=None, relations=None, confidence_band=None):
        """Return a predicate(edge_data: dict) -> bool applying AND semantics (D-11).
        When all three args are None, returned predicate accepts every edge (D-13 v1.12 byte-identity)."""
        band_range = _resolve_confidence_band(confidence_band)
        rel_filter = _validate_relations_filter_arg(relations)

        def predicate(edge_data):
            score = edge_data.get("confidence_score")
            if min_confidence is not None:
                if score is None or score < min_confidence:
                    return False
            if band_range is not None:
                if score is None:
                    return False
                lo, hi = band_range
                if not (lo <= score < hi):
                    return False
            if rel_filter is not None:
                if not rel_filter:                              # empty list = explicit zero-match
                    return False
                if edge_data.get("relation") not in rel_filter:
                    return False
            return True

        return predicate
    ```
