def _deep_merge_with_provenance(
    base: dict, override: dict, source_path: Path,
    provenance: dict[str, Path], _prefix: str = "",
) -> dict:
    result = base.copy()
    for key, value in override.items():
        dotted = f"{_prefix}{key}" if _prefix else key
        if key in result and isinstance(result[key], dict) and isinstance(value, dict):
            result[key] = _deep_merge_with_provenance(
                result[key], value, source_path, provenance, _prefix=f"{dotted}."
            )
        else:
            result[key] = value
            provenance[dotted] = source_path  # <-- SCALAR WRITE; Phase 56 changes to .setdefault([]).append(...)
    return result
