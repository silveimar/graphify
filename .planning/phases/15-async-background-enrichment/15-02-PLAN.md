    def _run_description_pass(
        G,
        out_dir: Path,
        snapshot_id: str,
        *,
        budget_remaining: int,
        dry_run: bool,
        alias_map: dict[str, str],
        routing_skip: set[str],

        results: dict[str, str] = {}
        tokens_used = 0
        llm_calls = 0

                tokens_used += 300

            llm_calls += 1
