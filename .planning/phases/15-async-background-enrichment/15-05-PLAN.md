<objective>
Close the lock-coordination loop between foreground `/graphify` (via `graphify run`) and background enrichment (via `graphify enrich` — Plans 01-03). Per ENRICH-07: foreground always wins — it acquires the `.enrichment.lock` before writing `graph.json`, SIGTERMs any active enrichment PID, and waits for clean release. Per ENRICH-06: `watch.py` gets an opt-in `--enrich` flag that spawns `graphify enrich` via `subprocess.Popen` after each rebuild; the watcher's `atexit` handler sends SIGTERM to the child on exit (Pitfall 4 mitigation). No auto-spawn on default path.

Purpose: Plans 01-04 made enrichment and overlay work in isolation; this plan makes them play nicely with the existing graphify pipeline. The hard invariant — foreground always wins, enrichment SIGTERM-aborts cleanly, no zombies — is structurally enforced here. This is the "peaceful coexistence" half of the lock protocol; without it, Pitfall 3 (graph.json overwrite) becomes possible via a timing race.

```python
# Near the existing `elif args.cmd == "run":` block in main(),
# wrap the to_json() call with an enrichment-lock acquisition.
# Current code (watch.py:78 for reference): to_json(G, communities, str(out / "graph.json"))

def _foreground_acquire_enrichment_lock(out_dir: Path, timeout_seconds: float = 30.0) -> int | None:
    """Acquire .enrichment.lock for a foreground rebuild. Foreground ALWAYS WINS:
    - If lock free: acquire (LOCK_EX) and return fd.
    - If lock held: read .enrichment.pid, send SIGTERM to PID, then block on LOCK_EX.
    - If pid unreadable or process already gone: fall back to blocking LOCK_EX (lock cleanup is kernel-driven).
    - Returns fd (caller must release), or None if out_dir doesn't exist (no lock needed — first-run).
    """
    import fcntl, os, signal as _signal, time, json as _json
    from graphify.enrich import LOCK_FILENAME, PID_FILENAME
    if not out_dir.exists():
        return None
    lock_path = out_dir / LOCK_FILENAME
    fd = os.open(str(lock_path), os.O_RDWR | os.O_CREAT, 0o644)
    try:
        fcntl.flock(fd, fcntl.LOCK_EX | fcntl.LOCK_NB)
        return fd  # free — acquired immediately
    except BlockingIOError:
        pass  # held by enrichment — SIGTERM it below
    # Read heartbeat PID; SIGTERM if process alive
    pid_path = out_dir / PID_FILENAME
    try:
        heartbeat = _json.loads(pid_path.read_text(encoding="utf-8"))
        pid = int(heartbeat.get("pid", 0))
        if pid > 0:
            try:
                os.kill(pid, _signal.SIGTERM)
                print(f"[graphify] foreground rebuild: sent SIGTERM to enrichment pid={pid}",
                      file=sys.stderr)
            except ProcessLookupError:
                pass  # already gone — lock will free on its own
    except (OSError, _json.JSONDecodeError, ValueError):
        pass  # no readable heartbeat — fall through to blocking wait
    # Block until the enrichment process releases the lock (SIGTERM handler calls LOCK_UN)
    deadline = time.time() + timeout_seconds
    while time.time() < deadline:
        try:
            fcntl.flock(fd, fcntl.LOCK_EX | fcntl.LOCK_NB)
            return fd
        except BlockingIOError:
            time.sleep(0.1)
    # Timeout escape: force-block (still bounded by outer caller's shell timeout)
    fcntl.flock(fd, fcntl.LOCK_EX)
    return fd


def _foreground_release_enrichment_lock(fd: int | None) -> None:
    """Unlock + close. Safe to call with None."""
    import fcntl, os
    if fd is None:
        return
    try:
        fcntl.flock(fd, fcntl.LOCK_UN)
    except OSError:
        pass
    os.close(fd)
```
