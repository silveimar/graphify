```python
def _load_existing_enrichment(out_dir: Path, snapshot_id: str) -> dict:
    """Read enrichment.json; return its `passes` sub-dict IFF:
      - File exists AND
      - Parses as JSON AND

def _validate_enrichment_envelope(data: object) -> bool:
    """Strict shape check — returns True only if:
      - isinstance(data, dict)
      - data.get("version") == 1
      - isinstance(data.get("snapshot_id"), str) and len(snapshot_id) > 0
      - isinstance(data.get("passes"), dict)

          description → dict[str, str]
          patterns    → list[dict] (each with pattern_id / nodes / summary)
          community   → dict[str, str] (keys are community-id-as-str)
          staleness   → dict[str, str] (values one of {FRESH, STALE, GHOST})

    Returns False on ANY deviation. No exceptions — log to stderr and return False
    so callers can fall through to "fresh run" gracefully.
    """
```

    ```python
    def _validate_enrichment_envelope(data: object) -> bool:
        """D-05 schema strict-check. Returns False on ANY deviation.
        Logs warnings to stderr when shape is rejected (for user debuggability).
        """
        if not isinstance(data, dict):
            return False
        if data.get("version") != 1:
            print(f"[graphify] enrichment: envelope version != 1 "
                  f"(got {data.get('version')!r}); discarding", file=sys.stderr)
            return False
        snapshot_id = data.get("snapshot_id")
        if not isinstance(snapshot_id, str) or not snapshot_id:
            return False
        passes = data.get("passes")
        if not isinstance(passes, dict):
            return False
        for pass_name in passes:
            if pass_name not in PASS_NAMES:
                print(f"[graphify] enrichment: unexpected pass key {pass_name!r} "
                      f"in v1 envelope; discarding", file=sys.stderr)
                return False
        # Per-pass value shape checks (lenient: only verify top-level container shape,
        # not every inner element — commit code validates on write).
        if "description" in passes and not isinstance(passes["description"], dict):
            return False
        if "patterns" in passes and not isinstance(passes["patterns"], list):
            return False
        if "community" in passes and not isinstance(passes["community"], dict):
            return False
        if "staleness" in passes and not isinstance(passes["staleness"], dict):
            return False
        if "staleness" in passes:
            valid_labels = {"FRESH", "STALE", "GHOST"}
            for nid, label in passes["staleness"].items():
                if label not in valid_labels:
                    return False
        return True
    ```

    ```python
    def _load_existing_enrichment(out_dir: Path, snapshot_id: str) -> dict:
        """D-07: return the existing envelope iff its snapshot_id matches and schema valid.
        Returns {} to signal 'fresh run'. Never raises — all errors become empty-return.

        p = out_dir / "enrichment.json"
        if not p.exists():
            return {}
        try:
            data = json.loads(p.read_text(encoding="utf-8"))
        except (OSError, json.JSONDecodeError) as exc:
            print(f"[graphify] enrichment: existing enrichment.json unreadable: {exc}; "
                  f"starting fresh", file=sys.stderr)
            return {}
        if not _validate_enrichment_envelope(data):
            return {}
        if data.get("snapshot_id") != snapshot_id:
            # D-07: snapshot mismatch → discard (caller will overwrite on first commit)
            return {}
        return data  # returns full envelope so caller can read `passes` dict
    ```
