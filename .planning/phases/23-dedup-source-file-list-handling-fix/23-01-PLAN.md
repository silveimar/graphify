```python
def _iter_sources(source_file: "str | list[str] | None") -> list[str]:
    if not source_file:
        return []
    if isinstance(source_file, str):
        return [source_file]
    if isinstance(source_file, list):
        return [s for s in source_file if isinstance(s, str) and s]
    return []
```
